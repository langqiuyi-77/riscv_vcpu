1//! Detect instruction sets (ISA extensions) by trap-and-return procedure
2//!
3//! First, it disables all S-level interrupts. Remaining traps in RISC-V core
4//! are all exceptions.
5//! Then, it filters out illegal instruction from exceptions.
6//! ref: <https://github.com/luojia65/zihai/blob/main/zihai/src/detect.rs>
7
8use core::arch::{asm, naked_asm};
9use riscv::register::{
10    scause::{Exception, Scause, Trap},
11    sstatus,
12    stvec::{self, Stvec, TrapMode},
13};
14
15/// Detect if hypervisor extension exists on current hart environment
16///
17/// This function tries to read hgatp and returns false if the read operation failed.
18pub fn detect_h_extension() -> bool {
19    // run detection by trap on csrr instruction.
20    let ans = with_detect_trap(0, || unsafe {
21        asm!("csrr  {}, 0x680", out(reg) _, options(nomem, nostack)); // 0x680 => hgatp
22    });
23    // return the answer from output flag. 0 => success, 2 => failed, illegal instruction
24    ans != 2
25}

27// Tries to execute all instructions defined in clojure `f`.
28// If resulted in an exception, this function returns its exception id.
29//
30// This function is useful to detect if an instruction exists on current environment.
31#[inline]
32fn with_detect_trap(param: usize, f: impl FnOnce()) -> usize {
33    // disable interrupts and handle exceptions only
34    let (sie, stvec, tp) = unsafe { init_detect_trap(param) };
35    // run detection inner
36    f();
37    // restore trap handler and enable interrupts
38    unsafe { restore_detect_trap(sie, stvec, tp) }
39}

41// rust trap handler for detect exceptions
42extern "C" fn rust_detect_trap(trap_frame: &mut TrapFrame) {
43    // store returned exception id value into tp register
44    // specially: illegal instruction => 2
45    trap_frame.tp = trap_frame.scause.bits();
46    // if illegal instruction, skip current instruction
47    match trap_frame.scause.cause() {
48        Trap::Exception(Exception::IllegalInstruction) => {
49            let mut insn_bits = riscv_illegal_insn_bits((trap_frame.stval & 0xFFFF) as u16);
50            if insn_bits == 0 {
51                let insn_half = unsafe { *(trap_frame.sepc as *const u16) };
52                insn_bits = riscv_illegal_insn_bits(insn_half);
53            }
54            // skip current instruction
55            trap_frame.sepc = trap_frame.sepc.wrapping_add(insn_bits);
56        }
57        Trap::Exception(_) => unreachable!(), // FIXME: unexpected instruction errors
58        Trap::Interrupt(_) => unreachable!(), // filtered out for sie == false
59    }
60}

62// Gets risc-v instruction bits from illegal instruction stval value, or 0 if unknown
63#[inline]
64fn riscv_illegal_insn_bits(insn: u16) -> usize {
65    if insn == 0 {
66        return 0; // stval[0..16] == 0, unknown
67    }
68    if insn & 0b11 != 0b11 {
69        return 2; // 16-bit
70    }
71    if insn & 0b11100 != 0b11100 {
72        return 4; // 32-bit
73    }
74    // FIXME: add >= 48-bit instructions in the future if we need to detect such instrucions
75    // >= 48-bit, unknown from this function by now
76    0
77}

79// Initialize environment for trap detection and filter in exception only
80#[inline]
81unsafe fn init_detect_trap(param: usize) -> (bool, Stvec, usize) {
82    // clear SIE to handle exception only
83    let stored_sie = sstatus::read().sie();
84    unsafe {
85        sstatus::clear_sie();
86    }
87    // use detect trap handler to handle exceptions
88    let stored_stvec = stvec::read();
89    let mut trap_addr = on_detect_trap as usize;
90    if trap_addr & 0b1 != 0 {
91        trap_addr += 0b1;
92    }
93    let stored_tp: usize;
94    unsafe {
95        stvec::write(trap_addr, TrapMode::Direct);
96        // store tp register. tp will be used to load parameter and store return value
97        asm!("mv  {}, tp", "mv  tp, {}", out(reg) stored_tp, in(reg) param, options(nomem, nostack));
98    }
99    // returns preserved previous hardware states
100    (stored_sie, stored_stvec, stored_tp)
101}
102
103// Restore previous hardware states before trap detection
104#[inline]
105unsafe fn restore_detect_trap(sie: bool, stvec: Stvec, tp: usize) -> usize {
106    // read the return value from tp register, and restore tp value
107    let ans: usize;
108    unsafe {
109        asm!("mv  {}, tp", "mv  tp, {}", out(reg) ans, in(reg) tp, options(nomem, nostack));
110        // restore trap vector settings
111        asm!("csrw  stvec, {}", in(reg) stvec.bits(), options(nomem, nostack));
112        // enable interrupts
113        if sie {
114            sstatus::set_sie();
115        };
116    }
117    ans
118}

122struct TrapFrame {
123    ra: usize,
124    tp: usize,
125    a0: usize,
126    a1: usize,
127    a2: usize,
128    a3: usize,
129    a4: usize,
130    a5: usize,
131    a6: usize,
132    a7: usize,
133    t0: usize,
134    t1: usize,
135    t2: usize,
136    t3: usize,
137    t4: usize,
138    t5: usize,
139    t6: usize,
140    sstatus: usize,
141    sepc: usize,

144}

146// Assembly trap handler for instruction detection.
147//
148// This trap handler shares the same stack from its prospective caller,
149// the caller must ensure it has abundant stack size for a trap handler.
150//
151// This function should not be used in conventional trap handling,
152// as it does not preserve a special trap stack, and it's designed to
153// handle exceptions only rather than interrupts.
154#[unsafe(naked)]
155unsafe extern "C" fn on_detect_trap() -> ! {
156    naked_asm!(
157        ".p2align 2",
158        "addi   sp, sp, -8*21",
159        "sd     ra, 0*8(sp)",
160        "sd     tp, 1*8(sp)",
161        "sd     a0, 2*8(sp)",
162        "sd     a1, 3*8(sp)",
163        "sd     a2, 4*8(sp)",
164        "sd     a3, 5*8(sp)",
165        "sd     a4, 6*8(sp)",
166        "sd     a5, 7*8(sp)",
167        "sd     a6, 8*8(sp)",
168        "sd     a7, 9*8(sp)",
169        "sd     t0, 10*8(sp)",
170        "sd     t1, 11*8(sp)",
171        "sd     t2, 12*8(sp)",
172        "sd     t3, 13*8(sp)",
173        "sd     t4, 14*8(sp)",
174        "sd     t5, 15*8(sp)",
175        "sd     t6, 16*8(sp)",
176        "csrr   t0, sstatus",
177        "sd     t0, 17*8(sp)",
178        "csrr   t1, sepc",
179        "sd     t1, 18*8(sp)",
180        "csrr   t2, scause",
181        "sd     t2, 19*8(sp)",
182        "csrr   t3, stval",
183        "sd     t3, 20*8(sp)",
184        "mv     a0, sp",
185        "call   {rust_detect_trap}",
186        "ld     t0, 17*8(sp)",
187        "csrw   sstatus, t0",
188        "ld     t1, 18*8(sp)",
189        "csrw   sepc, t1",
190        "ld     t2, 19*8(sp)",
191        "csrw   scause, t2",
192        "ld     t3, 20*8(sp)",
193        "csrw   stval, t3",
194        "ld     ra, 0*8(sp)",
195        "ld     tp, 1*8(sp)",
196        "ld     a0, 2*8(sp)",
197        "ld     a1, 3*8(sp)",
198        "ld     a2, 4*8(sp)",
199        "ld     a3, 5*8(sp)",
200        "ld     a4, 6*8(sp)",
201        "ld     a5, 7*8(sp)",
202        "ld     a6, 8*8(sp)",
203        "ld     a7, 9*8(sp)",
204        "ld     t0, 10*8(sp)",
205        "ld     t1, 11*8(sp)",
206        "ld     t2, 12*8(sp)",
207        "ld     t3, 13*8(sp)",
208        "ld     t4, 14*8(sp)",
209        "ld     t5, 15*8(sp)",
210        "ld     t6, 16*8(sp)",
211        "addi   sp, sp, 8*21",
212        "sret",
213        rust_detect_trap = sym rust_detect_trap,
214    )
